\begin{frame}
  \frametitle{The old way}
  
  \begin{itemize}
    \item Set up a server room
    \item Order servers
    \item Rack servers
    \item Configure servers
    \item Run servers
    \item Maintain all of the above 
  \end{itemize}
  
  Associated time scale: weeks
  

\end{frame}


\begin{frame}
  \frametitle{The new way}
  
  \begin{itemize}
    \item Set up an account with a cloud provider
    \item Start servers
    \item Run servers
  \end{itemize}
  
  Associated time scale: minutes

\end{frame}

\begin{frame}
  \frametitle{What does a public cloud provider do?}
  
  A public cloud provider supplies us with
  
  \begin{itemize}
    \item Compute
    \item Storage
    \item Network
  \end{itemize}
  
  resources on an on-demand basis.

\end{frame}

\begin{frame}
  \frametitle{What are some advantages to using a public cloud?}

  Compared to ``the old way'', cloud providers offer advantages in terms of 
  
	\begin{itemize}
    	\item Money
    	\item Time
    	\item Space
	    \item Location
	    \item Elastic scalability
  	\end{itemize}


\end{frame}

\begin{frame}
  \frametitle{Alternative classification of cloud services}
  
  Sometime we describe cloud services in terms of \emph{X as a Service} (Xaas)
  
  \begin{itemize}
    \item SaaS (Software as a Service)
    \item PaaS (Platform as a Service)
    \item IaaS (Infrastructure as a Service)
    \item AaaS (API as a Service)
    \item ...and so on
  \end{itemize}
  
  Can we think of examples of these or of other XaaS types?
\end{frame}
